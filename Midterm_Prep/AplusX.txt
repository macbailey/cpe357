#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/stat.h>
#include <sys/types.h>

void AplusX(char* filePath)
{
  struct stat sb; 
  if(stat(filePath, &sb) != -1)
  {
    if(S_ISREG(sb.st_mode)){
      if(sb.st_mode & S_IXUSR || sb.st_mode & S_IXGRP || sb.st_mode & S_IXOTH)
      {
        chmod(filePath, 111);  
      }
    }
  } 
}
int main(int argc, char* argv[])
{
  char* filePath = argv[1]; 
  AplusX(filePath);
  return 0; 
}
